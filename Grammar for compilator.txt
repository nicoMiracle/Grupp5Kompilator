function-name -> [a-zA-Z][a-zA-Z0-9]*
variable -> [a-zA-Z_][a-zA-Z0-9_]*
number -> [0-9]+
string -> ".*"

Stmt -> WhileStm | IfStm | AssignmStmt | Block
WhileStm -> while "(" Exp ")" Stmt
IfStm -> if "(" Exp ")" Stmt
Exp -> ID | MoreEq | LessEq | More | Less 
MoreEq ->  Exp ">=" Exp
LessEq ->  Exp "<=" Exp
Less ->  Exp "<" Exp
More ->  Exp ">" Exp
Block -> "{" Stmt* "}" 





LPAR = "("
RPAR = ")"
ID = [a-z][a-z0-9]*
IF = "if"
WHILE = "while"
PLUS = "+"
MINUS = "-"
